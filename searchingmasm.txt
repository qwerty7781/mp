DATA SEGMENT
STRING1 DB 11H, 22H, 33H, 44H,
55H
MSG_FOUND DB "FOUND$"
MSG_NOT_FOUND DB "NOT
FOUND$"
SE DB 33H
DATA ENDS
PRINT MACRO MSG
MOV AH, 09H
LEA DX, MSG
INT 21H
ENDM
CODE SEGMENT
ASSUME CS: CODE, DS: DATA
START:
MOV AX, DATA
MOV DS, AX
MOV AL, [SE]
LEA SI, STRING1
MOV CX, 05H
UP:
MOV BL, [SI]
CMP AL, BL
JE FOUND
INC SI
DEC CX
JNZ UP
PRINT MSG_NOT_FOUND
JMP END_PROGRAM
FOUND:
PRINT MSG_FOUND
END_PROGRAM:
MOV AX, 4C00H
INT 21H
CODE ENDS
END START
